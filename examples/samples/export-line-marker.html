Narwhal.export('lineMarker', function (data, layer) {
        var y = this.yScale(data);
        layer.append('path')
            .attr('d', 'M0,' + y + ' L' + this.options.chart.plotWidth + ',' + y)
            .attr('stroke', '#000');
    });

    new Narwhal({
            el: '.export-line-marker',
            xAxis: {
                title: 'Index',
            },
            yAxis: {
                smartAxis: false,
                title: 'Value'
            }
        })
        .cartesian()
        .line([1, 2, 3, 4, 5, 4, 3, 2, 1])
        .lineMarker(4)
        .lineMarker(1)
        .render();
