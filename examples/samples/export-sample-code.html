Narwhal.export('myExportedVisualization', function (data, layer) {
        layer.append('circle')
            .attr('cx', this.xScale(data))
            .attr('cy', this.yScale(data))
            .attr('r', 45);
    });

    new Narwhal({ el: '.export-sample-code'})
        .cartesian()
        .myExportedVisualization(3)
        .render();
